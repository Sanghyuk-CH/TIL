# closure에 대해 설명해 주시고 example이 있다면 이야기해 주세요

function makeAdder(x) {
var y = 1;
return function(z) {
y = 100;
return x + y + z;
};
}
var add5 = makeAdder(5);
var add10 = makeAdder(10);
//클로저에 x와 y의 환경이 저장됨
console.log(add5(2)); // 107 (x:5 + y:100 + z:2)
console.log(add10(2)); // 112 (x:10 + y:100 + z:2)
//함수 실행 시 클로저에 저장된 x, y값에 접근하여 값을 계산

# call apply bind 에 대해 설명해 보세요. 언제 다르게 쓰나요?

# deep 과 shallow object copy에 대해 설명해 주시고 용도가 있다면 말씀해 주세요

# Can you describe the main difference between a forEach loop and a .map() loop and why you would pick one versus the other?

forEach 는 각 요소를 제공된 함수로 한번씩 실행을 하는것이고
map은 각 요소가 제공된 함수를 거쳐 새로운 배열을 리턴합니다.
저는 map이 원본배열을 바꾸지 않고 새로운 배열을 리턴하는 점과 속도또한 빠르다는점을 생각하여 map 을 선택하겠습니다.

# javascript 의 이벤트루프는 무엇인가?

자바스크립트는 스크립트를 위에서부터 읽으면서 실행됩니다.
첫번째 스크립트를 읽었을때 콜스택으로 넘기고 비동기 함수일경우 콜스택에서 Web APIs로 넘어가고 이후 비동기 함수의 콜백함수를 콜백 큐로 넘깁니다.
이벤트 루프는 콜스택이 비었을때 콜백큐의 첫번째 콜백을 콜스택에 밀어넣습니다. 이러한 방식에 의해 싱글스레드인 자바스크립트가 멀티쓰레드처럼 보이게 됩니다.
콜백큐에는 세가지의 큐가 있고 우선순위로 나열하자면 Microtask Queue, Animation Frames, Task Queue가 됩니다. 프로미스와 then은 Microtask Queue에 담기게 되며
requestAnimationFrame API가 실행되면 Animation Frames에 담기게 되고, setTimeout은 Task queue에 담기게 되며 하나의 큐가 다 비워져야 다음 큐를 시작하는 실행을 하게 된다.

# 오버라이딩과 오버로딩

오버라이딩은 상위클래스의 메서드와 같은 시그니처(이름, 반환형, 인자)로 메소드를 재정의하는 것을 말한다. 오버로딩은 메소드이름만 같고 시그니처가 다른 것들을 말한다.
